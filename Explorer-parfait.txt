/* --------------------------
   Explorer-specific player controls (updated)
   - attachExplorerControls: used only for Explorer videos
   - smooth drag: video.currentTime is updated continuously while user drags
   - overlay show/hide global toggle on click in explorer media area
---------------------------*/
(function(){
  window.__explorerOverlaysVisible = true;
  window.__explorerControls = window.__explorerControls || [];

  function fmt(s){ s=Math.floor(s||0); const m=Math.floor(s/60); const sec=s%60; return m+':'+String(sec).padStart(2,'0'); }

  function toggleAllExplorerOverlays(force){
    if(typeof force === 'boolean') window.__explorerOverlaysVisible = force;
    else window.__explorerOverlaysVisible = !window.__explorerOverlaysVisible;
    window.__explorerControls.forEach(obj=>{
      try{
        obj.overlay.style.display = window.__explorerOverlaysVisible ? 'flex' : 'none';
      }catch(e){}
    });
  }

  function setupExplorerClickToggle(root){
    root.addEventListener('click', function(e){
      const v = e.target.closest && e.target.closest('.media-container');
      if(!v) return;
      toggleAllExplorerOverlays();
    }, {passive:true});
  }
  if(explorerEl) setupExplorerClickToggle(explorerEl);

  function attachExplorerControls(video, container){
    if(!video || !container) return;
    if(container._explorerAttached) return;
    container._explorerAttached = true;

    // remove existing overlay if any
    const existing = container.querySelector('.player-overlay');
    if(existing) existing.remove();

    const overlay = document.createElement('div');
    overlay.className = 'player-overlay';
    overlay.style.pointerEvents = 'none';
    overlay.style.display = window.__explorerOverlaysVisible ? 'flex' : 'none';
    overlay.style.flexDirection = 'column';
    overlay.style.justifyContent = 'flex-end';
    overlay.style.alignItems = 'stretch';

    const big = document.createElement('div');
    big.className = 'big-play';
    big.style.pointerEvents = 'auto';
    big.textContent = video.paused ? '▶' : '❚❚';

    const progressRow = document.createElement('div');
    progressRow.className = 'progress-row';
    progressRow.style.pointerEvents = 'auto';

    const timeLeft = document.createElement('div');
    timeLeft.className = 'time';
    timeLeft.textContent = '0:00';

    const seek = document.createElement('div');
    seek.className = 'seek';
    const fill = document.createElement('div');
    fill.className = 'fill';
    const thumb = document.createElement('div');
    thumb.className = 'thumb';
    // ensure touch-action none so browser doesn't steal gestures
    seek.style.touchAction = 'none';
    thumb.style.touchAction = 'none';
    seek.appendChild(fill);
    seek.appendChild(thumb);

    const timeRight = document.createElement('div');
    timeRight.className = 'time';
    timeRight.textContent = '0:00';

    progressRow.appendChild(timeLeft);
    progressRow.appendChild(seek);
    progressRow.appendChild(timeRight);

    overlay.appendChild(big);
    overlay.appendChild(progressRow);

    container.style.position = container.style.position || 'relative';
    container.appendChild(overlay);

    const obj = { video, container, overlay, big, seek, fill, thumb, timeLeft, timeRight };
    window.__explorerControls.push(obj);

    function updateUI(){
      const dur = video.duration || 0;
      const cur = video.currentTime || 0;
      timeLeft.textContent = fmt(cur);
      timeRight.textContent = fmt(dur);
      const pct = dur > 0 ? (cur / dur) * 100 : 0;
      fill.style.width = pct + '%';
      thumb.style.left = pct + '%';
      big.textContent = video.paused ? '▶' : '❚❚';
    }

    video.addEventListener('loadedmetadata', ()=>{ updateUI(); });
    video.addEventListener('timeupdate', ()=>{ updateUI(); });

    big.addEventListener('click', function(ev){
      ev.stopPropagation();
      try{
        if(video.paused) video.play().catch(()=>{});
        else video.pause();
      }catch(e){}
      updateUI();
    }, {passive:true});

    // click on seek jumps immediately
    seek.addEventListener('click', function(ev){
      ev.stopPropagation();
      const r = seek.getBoundingClientRect();
      const p = Math.max(0, Math.min(1, (ev.clientX - r.left) / r.width));
      try{ video.currentTime = p * (video.duration || 0); }catch(e){}
      updateUI();
    }, {passive:true});

    // pointer drag: simplified robust implementation
    let dragging = false;

    function onPointerDown(ev){
      ev.preventDefault();
      ev.stopPropagation();
      dragging = true;
      // update position immediately
      const r = seek.getBoundingClientRect();
      const x = Math.max(r.left, Math.min(r.right, ev.clientX));
      const p = (x - r.left) / r.width;
      try{ video.currentTime = p * (video.duration || 0); }catch(e){}
      updateUI();
    }

    function onPointerMove(ev){
      if(!dragging) return;
      ev.preventDefault();
      ev.stopPropagation();
      const r = seek.getBoundingClientRect();
      const x = Math.max(r.left, Math.min(r.right, ev.clientX));
      const p = (x - r.left) / r.width;
      try{ video.currentTime = p * (video.duration || 0); }catch(e){}
      updateUI();
    }

    function onPointerUp(ev){
      if(!dragging) return;
      ev.preventDefault();
      ev.stopPropagation();
      dragging = false;
      // final update
      updateUI();
    }

    // attach pointer listeners
    thumb.addEventListener('pointerdown', onPointerDown, {passive:false});
    seek.addEventListener('pointerdown', onPointerDown, {passive:false});
    window.addEventListener('pointermove', onPointerMove, {passive:false});
    window.addEventListener('pointerup', onPointerUp, {passive:false});
    window.addEventListener('pointercancel', onPointerUp, {passive:false});

    container.addEventListener('click', function(e){ e.stopPropagation(); }, {passive:true});
    video.addEventListener('click', function(e){ e.stopPropagation(); }, {passive:true});

    video.addEventListener('play', ()=>{ updateUI(); if(window.__explorerOverlaysVisible) overlay.style.display='flex'; });
    video.addEventListener('pause', ()=>{ updateUI(); if(window.__explorerOverlaysVisible) overlay.style.display='flex'; });

    setTimeout(()=> updateUI(), 30);
  }

  window.attachExplorerControls = attachExplorerControls;

  const mo = new MutationObserver((mutations)=>{
    for(const m of mutations){
      for(const n of m.addedNodes){
        if(!(n instanceof HTMLElement)) continue;
        const vids = n.matches && n.matches('video') ? [n] : Array.from(n.querySelectorAll ? n.querySelectorAll('video') : []);
        vids.forEach(v=>{
          const container = v.parentElement || n;
          setTimeout(()=>{ attachExplorerControls(v, container); }, 8);
        });
      }
    }
  });
  if(explorerEl) mo.observe(explorerEl, { childList:true, subtree:true });

  if(explorerEl){
    explorerEl.querySelectorAll('video').forEach(v=>{
      const container = v.parentElement || explorerEl;
      setTimeout(()=>{ attachExplorerControls(v, container); }, 8);
    });
  }

  window.toggleExplorerOverlays = toggleAllExplorerOverlays;
})();
